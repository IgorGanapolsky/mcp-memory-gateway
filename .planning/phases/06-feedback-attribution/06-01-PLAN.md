// Attribution side-effects — fire-and-forget, never throw
try {
  const toolName = event.toolName || event.tool_name || 'unknown';
  const toolInput = event.context || event.input || '';
  recordAction(toolName, toolInput);
  if (String(event.signal || '').toLowerCase().includes('negative') || String(event.signal || '').toLowerCase().includes('down')) {

  }
} catch (e) {
  // attribution is non-blocking
}
