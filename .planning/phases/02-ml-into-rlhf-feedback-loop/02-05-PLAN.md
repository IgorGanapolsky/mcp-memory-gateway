3. Time-decay verification:
```bash
node -e "
const { timeDecayWeight, HALF_LIFE_DAYS, DECAY_FLOOR } = require('./scripts/thompson-sampling');
const now = timeDecayWeight(new Date().toISOString());
const week = timeDecayWeight(new Date(Date.now() - 7*24*60*60*1000).toISOString());
const old = timeDecayWeight(new Date(Date.now() - 30*24*60*60*1000).toISOString());
console.log('Age 0d weight:', now.toFixed(6));
console.log('Age 7d weight:', week.toFixed(6));
console.log('Age 30d weight:', old.toFixed(6));
console.log('Floor:', DECAY_FLOOR, 'Half-life:', HALF_LIFE_DAYS, 'days');
"
```

4. Sequence and diversity verification (via test tmpdir):
```bash
node -e "
const { captureFeedback, getFeedbackPaths } = require('./scripts/feedback-loop');
const fs = require('fs'), path = require('path'), os = require('os');
const tmp = fs.mkdtempSync(path.join(os.tmpdir(), 'rlhf-proof-'));
process.env.RLHF_FEEDBACK_DIR = tmp;
captureFeedback({ signal:'positive', context:'test proof run', whatWorked:'evidence captured', tags:['testing'] });
captureFeedback({ signal:'negative', context:'debugging evidence', whatWentWrong:'error observed', whatToChange:'fix applied', tags:['debugging'] });
const seqPath = path.join(tmp, 'feedback-sequences.jsonl');
const divPath = path.join(tmp, 'diversity-tracking.json');
const seqLines = fs.existsSync(seqPath) ? fs.readFileSync(seqPath,'utf-8').trim().split('\n').filter(Boolean) : [];
const div = fs.existsSync(divPath) ? JSON.parse(fs.readFileSync(divPath,'utf-8')) : {};
console.log('Sequence entries:', seqLines.length);
if (seqLines.length) {
  const entry = JSON.parse(seqLines[0]);
  console.log('Sequence entry schema:', Object.keys(entry).join(', '));
  console.log('targetReward:', entry.targetReward);
  console.log('rewardSequence:', entry.features.rewardSequence);
}
console.log('Diversity score:', div.diversityScore);
console.log('Domains:', Object.keys(div.domains || {}));
delete process.env.RLHF_FEEDBACK_DIR;
fs.rmSync(tmp, { recursive: true });
"
```
